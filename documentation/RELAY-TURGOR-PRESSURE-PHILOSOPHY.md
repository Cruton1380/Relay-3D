  produceAttestation(anchor) {

      hash: sha256(snapshot),

  chainAttestation(attestation) {
    const prev_hash = this.getLastCommit().hash
    
    const commit = {
      attestation,
      prev_hash,           // Links to previous commit
      commit_hash: sha256(attestation + prev_hash),
      commit_index: this.getNextIndex()
    }

    return commit
  }
